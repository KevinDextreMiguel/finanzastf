Algoritmo CalcularValorFuturo
	Definir  VF,VA,tasa Como Real
	Definir tipoTasa Como Caracter
	Definir  m,n Como Real
	
	Leer VA
	Leer tasa
	Leer tipoTasa
	Leer n
	
	Si tipoTasa='N' Entonces
		Leer m
		VF = VA * ((1+tasa/m)^n)
	SiNo
		VF = VA * ((1+tasa)^n)
	Fin Si
FinAlgoritmo

Algoritmo CalcularCuota
	Definir cuota,monto,tasa Como Real
	Definir cantMeses Como Real
	
	Leer monto
	Leer tasa
	Leer cantMeses
	
	cuota = monto*((tasa*((1+tasa)^cantMeses))/(((1+tasa)^cantMeses)-1))
FinAlgoritmo




Algoritmo ConvertirNominalEfectiva
	Definir tasNominal,tasaEfectiva Como Real
	Definir m,n Como Entero
	
	Leer tasNominal
	Leer n
	Leer  m
	
	tasaEfectiva = (1+(tasNominal/m)^n)-1
FinAlgoritmo




Algoritmo ConvertirEfectivaNominal
	Definir tasaEfectiva,tasaNominal Como Real
	Definir m,n Como Entero
	
	Leer tasNominal
	Leer n
	Leer  m
	
	tasaNominal = m*(((1+tasaEfectiva)^(1/n))-1)
FinAlgoritmo




Algoritmo TaMoratoria
	Definir tasaMoratoria,montoMora,monto Como Real
	Definir diasTranscurrido Como Real
	Definir vencimiento Como Logico
	
	Leer vencimiento
	
	Si vencimiento Entonces
		Leer tasaMoratoria
		Leer monto
		Leer diasTranscurrido
		montoMora = tasaMoratoria*diasTranscurrido*monto
		Escribir  "Monto de mora: ",montoMora
	SiNo
		Escribir  "No hay tasa moratoria"
	Fin Si
FinAlgoritmo



Algoritmo ControlarLimite
	Definir LimiteCredito,MontoPrestamo,acumular Como Real	
	acumular=0
	
	Escribir "Limite Credito: "
	Leer LimiteCredito
	
	Repetir
		Escribir "Monto Prestamo: "
		Leer MontoPrestamo
		Si acumular+MontoPrestamo<=LimiteCredito Entonces
			acumular = acumular + MontoPrestamo
		SiNo
			Escribir "No procede, porque ya supera el límite"
		Fin Si
	Hasta Que acumular>=LimiteCredito
	Escribir "LLegaste al límite de crédito"
FinAlgoritmo
